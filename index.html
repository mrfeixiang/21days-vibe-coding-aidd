Claude Code가 생성한 코드 from rdkit import Chem from rdkit.Chem import Descriptors def calc_properties(smiles: str) -> dict: """SMILES로부터 기본 분자 성질 계산""" mol = Chem.MolFromSmiles(smiles) if mol is None: raise ValueError(f"유효하지 않은 SMILES: {smiles}") return { "MW": round(Descriptors.MolWt(mol), 2), "LogP": round(Descriptors.MolLogP(mol), 2), "HBD": Descriptors.NumHDonors(mol), "HBA": Descriptors.NumHAcceptors(mol), } # 테스트: Target A 억제제 후보 result = calc_properties("c1ccc2c(c1)c1ccnc3ccccc3c1[nH]2") print(result)
